Multiselect listen to change events (watch console logs)

        <option value="België">België</option>
        <option value="Duitsland">Duitsland</option>
        <option value="Frankrijk">Frankrijk</option>
      </vl-multiselect>
      <script type="text/javascript">
        document.getElementById('change-listener').addEventListener('change', (e) => {
          console.dir(`e.detail.value => ${e.detail.value}`);
          console.dir(`e.target.value => ${e.target.value}`);
        });
      </script>

Multiselect wordt manueel gedressed

      <script type="text/javascript">
        const laadDynamisch = () => {
          document.getElementById('vl-multiselect-dynamisch').dress({
            callbackFn: (select) => {
              return [
                {value: 'One', label: 'Label One', disabled: true},
                {value: 'Two', label: 'Label Two', selected: true},
                {value: 'Three', label: 'Label Three'},
              ];
            }
          });
        }
      </script>
      <button onclick="laadDynamisch()" type="button">Activeer dynamische data</button>
      <br/>
      <br/>
      <vl-multiselect id="vl-multiselect-dynamisch"></vl-multiselect>


Multiselect methoden

    <script type="text/javascript">
      const multiselect = () => {
        return document.getElementById('vl-multiselect-methoden');
      };

      const enable = () => {
        multiselect().enable();
      };

      const disable = () => {
        multiselect().disable();
      };

      const setValueByChoice = () => {
        multiselect().setValueByChoice('Duitsland');
      };

      const clearSelectedOptions = () => {
        multiselect().clearSelectedOptions();
      };

      const removeActive = () => {
        multiselect().removeActive();
      };

      const showDropdown = () => {
        multiselect().showDropdown();
      };

      const hideDropdown = () => {
        multiselect().hideDropdown();
      };

    </script>
    <br />
    <button onclick="enable()" type="button">enable()</button>
    <button onclick="disable()" type="button">disable()</button>
    <button onclick="setValueByChoice()" type="button">setValueByChoice('Duitsland')</button>
    <button onclick="clearSelectedOptions()" type="button">clearSelectedOptions()</button>
    <button onclick="removeActive()" type="button">removeActive()</button>
    <button onclick="showDropdown()" type="button">showDropdown()</button>
    <button onclick="hideDropdown()" type="button">hideDropdown()</button>
    <br />
    <br />
    <vl-multiselect id="vl-multiselect-methoden" data-vl-multiselect>
      <option value="België">België</option>
      <option value="Duitsland">Duitsland</option>
      <option value="Frankrijk">Frankrijk</option>
      <option value="Nederland">Nederland</option>

    </vl-multiselect>
