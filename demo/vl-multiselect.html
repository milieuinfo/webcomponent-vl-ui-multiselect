vl-ui-multiselect demo

Meer voorbeelden en documentatie.

Multiselect met meerdere opties

      
        <vl-multiselect>
          <option value="België">België</option>
          <option value="Duitsland">Duitsland</option>
          <option value="Frankrijk">Frankrijk</option>
        </vl-multiselect>

Multiselect met een voorgeselecteerde optie

      
        <vl-multiselect>
          <option selected value="Brugge">Brugge</option>
          <option value="Brussel">Brussel</option>
          <option value="Gent">Gent</option>
        </vl-multiselect>

Multiselect met gegroepeerde opties

      
        <vl-multiselect>
          <optgroup label="Landen">
            <option selected value="België">België</option>
            <option value="Duitsland">Duitsland</option>
            <option value="Frankrijk">Frankrijk</option>
          </optgroup>
          <optgroup label="Steden">
            <option selected value="Brugge">Brugge</option>
            <option value="Brussel">Brussel</option>
            <option value="Gent">Gent</option>
          </optgroup>
        </vl-multiselect>

Multiselect met error

Multiselect listen to change events

      
    <vl-multiselect id="change-listener">
      <option value="België">België</option>
      <option value="Duitsland">Duitsland</option>
      <option value="Frankrijk">Frankrijk</option>
    </vl-multiselect>
    <script type="text/javascript">
      document.getElementById('change-listener').addEventListener('change', (e) => {
        console.dir(e);
      });
    </script>
