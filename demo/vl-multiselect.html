Multiselect methoden

      <script type="text/javascript">
      <select id="vl-select-methoden"
              is="vl-select"
              data-vl-select-deletable>
        <option placeholder value="">Kies een land</option>
        <option value="België">België</option>
        <option value="Duitsland">Duitsland</option>
        <option value="Frankrijk">Frankrijk</option>
        <option value="Nederland">Nederland</option>
        <option value="Italië">Italië</option>
      </select>
  
      <button is="vl-button" onclick="document.getElementById('vl-select-methoden').dress()" type="button">Dress</button>
      <button is="vl-button" onclick="document.getElementById('vl-select-methoden').undress()" type="button">Undress</button>
      <button is="vl-button" onclick="document.getElementById('vl-select-methoden').enable()" type="button">Enable</button>
      <button is="vl-button" onclick="document.getElementById('vl-select-methoden').disable()" type="button">Disable</button>
      <button is="vl-button" onclick="document.getElementById('vl-select-methoden').setValueByChoice('Duitsland')" type="button">Kies Duitsland</button>
      <button is="vl-button" onclick="document.getElementById('vl-select-methoden').removeActive()" type="button">Verwijder selectie</button>
      <button is="vl-button" onclick="document.getElementById('vl-select-methoden').showDropdown()" type="button">Show dropdown</button>
      <button is="vl-button" onclick="document.getElementById('vl-select-methoden').hideDropdown()" type="button">Hide dropdown</button>
