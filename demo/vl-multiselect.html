Multiselect luister naar change events

      
        <select id="change-listener" is="vl-multiselect" multiple>
          <option value="België">België</option>
          <option value="Duitsland">Duitsland</option>
          <option value="Frankrijk">Frankrijk</option>
        </select>
        <script type="text/javascript">
          document.getElementById('change-listener').addEventListener('change', (e) => {
            console.dir(`e.detail.value => ${e.detail.value}`);
            console.dir(`e.target.value => ${e.target.value}`);
          });
        </script>

Multiselect enable en disable methode

      
        <select id="vl-multiselect-enable-disable-methode" is="vl-multiselect" multiple>
          <option value="België">België</option>
          <option value="Duitsland">Duitsland</option>
          <option value="Frankrijk">Frankrijk</option>
          <option value="Nederland">Nederland</option>
          <option value="Italië">Italië</option>
        </select>
  
        <button is="vl-button" onclick="document.getElementById('vl-multiselect-enable-disable-methode').enable()" type="button">Enable</button>
        <button is="vl-button" onclick="document.getElementById('vl-multiselect-enable-disable-methode').disable()" type="button">Disable</button>
